📖 题目

有n辆车大甩卖，第i辆车售价a[i]元。有m个人带着现金来申请购买，第i个到现场的人带的现金为b[i]元，只能买价格不超过其现金额的车子。你是大卖场总经理，希望将车和买家尽量多地进行一对一配对，请问最多卖出多少辆车？

❌ 原本错的代码

#include <iostream>
#include <cstdio>
#include <algorithm>
const int max_array = 200005;
using namespace std;
int main()
{
    freopen("car2.in", "r", stdin);
    freopen("car2.out", "w", stdout);
    int car_prices[max_array];
    int people_price[max_array];
    int how_many_cars = 0;
    int how_many_people = 0;
    int boundary_flag = 0;
    int ans = 0;
    cin >> how_many_cars >> how_many_people;
    for (size_t i = 0; i < how_many_cars; i++)
    {
        cin >> car_prices[i];
    }
    for (size_t i = 0; i < how_many_people; i++)
    {
        cin >> people_price[i];
    }
    sort(car_prices, car_prices + how_many_cars);
    sort(people_price, people_price + how_many_people);
    for (size_t i = 0; i < how_many_cars; i++)
    {
        //从最便宜的那辆车选其，挨个挨个看人有没有足够钱
        bool reached_end = false; //循环内部代码终止循环时会将这只flag更换为false，意思是这么贵的车，最富的人也买不起
        while (true)
        {
            if (boundary_flag >= how_many_people)
            {
                reached_end = true;
                break;
            }
            boundary_flag++;
            if (people_price[boundary_flag] >= car_prices[i])
            {
                ans++;
                break;
            }
        }
        if (reached_end)
        {
            break;
        }
    }
    cout << ans;
    return 0;
}

🐎 为神马错

  1. 变量boundary_flag不能在一开始就增加1，这样一开始0号元素就被忽略了，直接跳到第1号元素
  2. 第33行的if语句不应该是how_many_people，而应该是how_many_people-1，不然又回数组越界

⭕️ 正确代码

#include <iostream>
#include <cstdio>
#include <algorithm>
const int max_array = 200005;
using namespace std;
int main()
{
    freopen("car2.in", "r", stdin);
    freopen("car2.out", "w", stdout);
    int car_prices[max_array];
    int people_price[max_array];
    int how_many_cars = 0;
    int how_many_people = 0;
    int boundary_flag = 0;
    int ans = 0;
    cin >> how_many_cars >> how_many_people;
    for (size_t i = 0; i < how_many_cars; i++)
    {
        cin >> car_prices[i];
    }
    for (size_t i = 0; i < how_many_people; i++)
    {
        cin >> people_price[i];
    }
    sort(car_prices, car_prices + how_many_cars);
    sort(people_price, people_price + how_many_people);
    for (size_t i = 0; i < how_many_cars; i++)
    {
        //从最便宜的那辆车选其，挨个挨个看人有没有足够钱
        bool reached_end = false; //循环内部代码终止循环时会将这只flag更换为false，意思是这么贵的车，最富的人也买不起
        while (true)
        {
            if (boundary_flag >= how_many_people-1)
            {
                reached_end = true;
                break;
            }
            if (people_price[boundary_flag] >= car_prices[i])
            {
                ans++;
                break;
            }
            boundary_flag++;
        }
        if (reached_end)
        {
            break;
        }
    }
    cout << ans;
    return 0;
}
